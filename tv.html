// Paste link and PLAY button function async function play() { var qry = document.getElementById("url-input").value.trim(); if (qry === "") { if (!navigator.clipboard) { console.error('Clipboard API not supported'); return; } try { const text = await navigator.clipboard.readText(); if (text.trim() === "") { alert("Please Enter URL First"); return; } document.getElementById('url-input').value = text.trim(); qry = text.trim(); } catch (err) { console.error('Failed to read clipboard contents:', err); if (err.name === 'NotAllowedError') { console.error('Permission to read clipboard denied'); } return; } } document.getElementById("video").src = qry; document.getElementById("video").play(); }
